import pandas as pd

df = pd.read_excel(r"C:\MMD Lab\data.xlsx")

def mapper(data):
    mapped_data = []
    for age in data["age"]:
        if not pd.isna(age):
            mapped_data.append(age)
    return mapped_data

def reducer(mapped_data):
    total_age = 0
    count = 0
    for age in mapped_data:
        total_age += age
        count += 1
    return total_age / count

mapped = mapper(df)
average_age = reducer(mapped)

print("Average Age of Titanic Passengers:", round(average_age, 2))
